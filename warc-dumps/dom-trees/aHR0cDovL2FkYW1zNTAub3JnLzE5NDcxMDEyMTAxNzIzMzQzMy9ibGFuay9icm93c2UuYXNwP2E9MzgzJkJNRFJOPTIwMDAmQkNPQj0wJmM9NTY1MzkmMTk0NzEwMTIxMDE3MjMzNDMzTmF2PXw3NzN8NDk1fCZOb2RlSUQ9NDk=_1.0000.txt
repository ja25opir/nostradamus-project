        function ReadCookie( check_name ) {
	        // first we'll split this cookie up into name/value pairs
	        // note: document.cookie only returns name=value, not the other components
	        var a_all_cookies = document.cookie.split( ';' );
	        var a_temp_cookie = '';
	        var cookie_name = '';
	        var cookie_value = '';
	        var b_cookie_found = false; // set boolean t/f default f
        	
	        for ( i = 0; i < a_all_cookies.length; i++ )
	        {
		        // now we'll split apart each name=value pair
		        a_temp_cookie = a_all_cookies[i].split( '=' );
        		
        		
		        // and trim left/right whitespace while we're at it
		        cookie_name = a_temp_cookie[0].replace(/^\s+|\s+$/g, '');
        	
		        // if the extracted name matches passed check_name
		        if ( cookie_name == check_name )
		        {
			        b_cookie_found = true;
			        // we need to handle case where cookie has no value but exists (no = sign, that is):
			        if ( a_temp_cookie.length > 1 )
			        {
				        cookie_value = unescape( a_temp_cookie[1].replace(/^\s+|\s+$/g, '') );
			        }
			        // note that in cases where cookie is initialized but no value, null is returned
			        return cookie_value;
			        break;
		        }
		        a_temp_cookie = null;
		        cookie_name = '';
	        }
	        if ( !b_cookie_found )
	        {
		        return null;
	        }
        }

<div>"Chris dedicates his life to making a difference for kids. As our Discovery teacher, Chris has connected personally with virtually every student in our building. The students love his class and he makes a difference every day.&nbsp; Let me take this opportunity to provide a few examples about how he impacts students' lives.<br>&nbsp;<br>As part of the communications module, Chris teaches students how to communicate with adults.&nbsp; So much so, he has successfully assisted many students in addressing issues at home in a respectful and meaningful way. The students learn the skills and gain the confidence to discuss hard issues with their parents. In return, the students begin to trust and learn the skills that will take them through the rest of their lives.&nbsp; Chris will be the teacher that many students here at HLHS will remember in the future as the one that made a difference.&nbsp; Chris has great empathy for his students and the culture of District 50.&nbsp; He has the ability to work with students to re-engage in school and make Hidden Lake a success.&nbsp; His skills are reflected in the success of Hidden Lake High School."</div>
